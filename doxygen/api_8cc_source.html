61 void ApiInterface::insertRow(apifPtr re, int64_t recmd, void *reptr,
62  int64_t tableid)
63 {
64  setReEntry(re, recmd, reptr);
65 
66  if (transactionPtr->pendingcmd != NOCOMMAND)
67  {
68  transactionPtr->reenter(APISTATUS_PENDING);
69  return;
70  }
71 
72  uuRecord_s uur = { -1, -1, -1 };
73  transactionPtr->returnNewRow = uur; // why this here?
74  transactionPtr->pendingcmdid = transactionPtr->getnextpendingcmdid();
75  transactionPtr->pendingcmd = INSERT;
76  transactionPtr->zeroCurrentCmdState();
77  transactionPtr->currentCmdState.tableid = tableid;
78  transactionPtr->currentCmdState.tablePtr =
79  transactionPtr->schemaPtr->tables[tableid];
80 
81  indexInfo_s blankIdxInfo = {};
82  transactionPtr->currentCmdState.indexEntries.resize
83  (transactionPtr->currentCmdState.tablePtr->fields.size(), blankIdxInfo);
84  indexEntry_s blankIdxEntry = {};
85  transactionPtr->currentCmdState.rowidsEngineids.resize
86  (transactionPtr->currentCmdState.tablePtr->fields.size(), blankIdxEntry);
87 
88  for (size_t n=0; n<transactionPtr->currentCmdState.tablePtr->fields.size();
89  n++)
90  {
91  if (transactionPtr->checkNullConstraintOK(n)==false)
92  {
93  transactionPtr->reenter(APISTATUS_NULLCONSTRAINT);
94  return;
95  }
96 
97  if (transactionPtr->currentCmdState.tablePtr->fields[n].indextype != NONE)
98  {
99  transactionPtr->currentCmdState.indexEntries[n].engineid =
100  transactionPtr->getEngineid(transactionPtr->currentCmdState.tablePtr, n);
101  transactionPtr->currentCmdState.indexEntries[n].fieldid = n;
102  transactionPtr->currentCmdState.indexEntries[n].tableid =
103  transactionPtr->currentCmdState.tableid;
104  transactionPtr->currentCmdState.indexEntries[n].fieldVal =
105  transactionPtr->fieldValues[n];
106  }
107  else
108  {
109  transactionPtr->currentCmdState.indexEntries[n].engineid = -1;
110  }
111 
112  transactionPtr->currentCmdState.rowidsEngineids[n].engineid =
113  transactionPtr->currentCmdState.indexEntries[n].engineid;
114  }
115 
116  transactionPtr->currentCmdState.rowEngineid =
117  transactionPtr->getEngineid(transactionPtr->currentCmdState.tablePtr, 0);
118 
119  if (transactionPtr->currentCmdState.rowEngineid == -1)
120  {
121  transactionPtr->reenter(APISTATUS_NOTOK);
122  return;
123  }
124 
125  // now, prepare message for Engine, which expects:
126  // transaction_enginecmd NEWROW, transaction_messageStruct.payloadtype
127  // SUBTRANSACTIONCMDPAYLOAD
128  // reentrypoint 1
129  // tableid, row
130  // expects in return: rowid
131  class MessageSubtransactionCmd *msg = new class MessageSubtransactionCmd();
132  class MessageSubtransactionCmd &msgref = *msg;
133  msgref.subtransactionStruct.tableid=tableid;
134 
135  if (transactionPtr->currentCmdState.tablePtr->
136  makerow(&transactionPtr->fieldValues, &msgref.row)==false)
137  {
138  transactionPtr->reenter(APISTATUS_FIELD);
139  delete msg;
140  return;
141  }
142 
143  vector<fieldValue_s> f;
144  transactionPtr->currentCmdState.tablePtr->unmakerow(&msgref.row, &f);
145  transactionPtr->currentCmdState.row = msgref.row;
146  transactionPtr->sendTransaction(NEWROW, PAYLOADSUBTRANSACTION, 1,
147  transactionPtr->currentCmdState.rowEngineid,
148  (void *)msg);
149 }

151 void ApiInterface::deleteRow(apifPtr re, int64_t recmd, void *reptr,
152  uuRecord_s &uur)
153 {
154  setReEntry(re, recmd, reptr);
155 
156  if (!transactionPtr->stagedRows.count(uur))
157  {
158  transactionPtr->reenter(APISTATUS_NOTOK);
159  return;
160  }
161 
162  if (transactionPtr->stagedRows[uur].locktype != WRITELOCK ||
163  transactionPtr->stagedRows[uur].cmd != NOCOMMAND ||
164  transactionPtr->pendingcmd != NOCOMMAND)
165  {
166  transactionPtr->reenter(APISTATUS_NOTOK);
167  return;
168  }
169 
170  transactionPtr->pendingcmdid = transactionPtr->getnextpendingcmdid();
171  transactionPtr->pendingcmd = DELETE;
172  transactionPtr->zeroCurrentCmdState();
173 
174  transactionPtr->currentCmdState.originaluur = uur;
175 
176  // tableid,rowid,DELETEROW,SUBTRANSACTIONCMDPAYLOAD, returns status
177  class MessageSubtransactionCmd *msg = new class MessageSubtransactionCmd();
178  class MessageSubtransactionCmd &msgref = *msg;
179  msgref.subtransactionStruct.rowid =
180  transactionPtr->currentCmdState.originaluur.rowid;
181  msgref.subtransactionStruct.tableid =
182  transactionPtr->currentCmdState.originaluur.tableid;
183  transactionPtr->sendTransaction(DELETEROW, PAYLOADSUBTRANSACTION, 1,
184  transactionPtr->currentCmdState.originaluur.engineid,
185  (void *)msg);
